#include <stdio.h>

int main() {
    long long binary;
    int decimal = 0, base = 1, remainder;

    // Accept binary input from the user
    printf("Enter a binary number (composed of 0s and 1s): ");
    scanf("%lld", &binary);

    long long temp = binary; // Preserve original input for the final output

    // Conversion loop
    while (temp > 0) {
        remainder = temp % 10;       // Extract the rightmost bit (0 or 1)
        decimal += remainder * base; // Add positional value to the total
        base *= 2;                   // Shift positional value to the next power of 2
        temp /= 10;                  // Truncate the processed bit
    }

    printf("The binary number %lld in decimal is: %d\n", binary, decimal);

    return 0;
}
