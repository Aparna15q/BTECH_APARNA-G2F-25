#include <stdio.h>
#include <string.h>

int main() {
    int n;
    char words[100][100];
    char temp[100];

    // Get the number of words from the user
    printf("Enter the number of words: ");
    if (scanf("%d", &n) != 1 || n <= 0) {
        printf("Invalid input.\n");
        return 1;
    }

    // Input the words
    printf("Enter %d words:\n", n);
    for (int i = 0; i < n; i++) {
        scanf("%s", words[i]);
    }

    // Bubble Sort based on the length of strings
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            // Compare lengths using strlen()
            if (strlen(words[j]) > strlen(words[j + 1])) {
                // Swap the words if the current one is longer than the next
                strcpy(temp, words[j]);
                strcpy(words[j], words[j + 1]);
                strcpy(words[j + 1], temp);
            }
        }
    }

    // Output the sorted result
    printf("\nWords sorted by length (shortest to longest):\n");
    for (int i = 0; i < n; i++) {
        printf("%s\n", words[i]);
    }

    return 0;
}
