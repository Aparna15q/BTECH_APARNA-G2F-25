#include <stdio.h>

int main() {
    int rows = 5;

    // Outer loop handles the rows
    for (int i = 1; i <= rows; i++) {
        // Start from character 'A' at the beginning of each row
        char ch = 'A';

        // Inner loop handles characters in each row
        for (int j = 1; j <= i; j++) {
            printf("%c", ch);
            ch++; // Move to the next alphabet
        }
        // Move to the next line
        printf("\n");
    }
    return 0;
}
