#include <stdio.h>

// Function to calculate x^n using Binary Exponentiation
double calculate_power(double x, int n) {
    double result = 1.0;
    long long abs_n = n; // Use long long to prevent overflow if n is INT_MIN

    // Handle negative exponents: x^(-n) = 1 / (x^n)
    if (abs_n < 0) {
        abs_n = -abs_n;
        x = 1.0 / x;
    }

    // Binary exponentiation loop O(log n)
    while (abs_n > 0) {
        // If the current bit of the exponent is odd, multiply the result
        if (abs_n % 2 == 1) {
            result *= x;
        }
        // Square the base and halve the exponent
        x *= x;
        abs_n /= 2;
    }

    return result;
}

int main() {
    double base;
    int exponent;

    printf("Enter base (x): ");
    if (scanf("%lf", &base) != 1) {
        printf("Invalid base input.\n");
        return 1;
    }

    printf("Enter integer exponent (n): ");
    if (scanf("%d", &exponent) != 1) {
        printf("Invalid exponent input.\n");
        return 1;
    }

    double final_result = calculate_power(base, exponent);

    printf("%.4f raised to the power of %d is: %.6f\n", base, exponent, final_result);

    return 0;
}
