#include <stdio.h>

int main() {
    int rows, i, j;

    // Get input from user
    printf("Enter the number of rows: ");
    scanf("%d", &rows);

    // Outer loop for the number of rows
    for (i = 1; i <= rows; i++) {

        // Inner loop to print the row number repeatedly
        for (j = 1; j <= i; j++) {
            printf("%d ", i);
        }

        // Move to the next line after finishing each row
        printf("\n");

 }

    return 0;
}
