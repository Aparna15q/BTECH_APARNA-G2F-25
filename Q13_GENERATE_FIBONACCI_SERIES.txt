#include <stdio.h>

int main() {
    int terms;
    // Use long long to handle larger Fibonacci numbers without overflow
    unsigned long long first = 0, second = 1, next;

    printf("Enter the number of terms: ");
    if (scanf("%d", &terms) != 1 || terms <= 0) {
        printf("Please enter a valid positive integer.\n");
        return 1;
    }

    printf("Fibonacci Series: ");

    for (int i = 1; i <= terms; ++i) {
        // Print the current term
        printf("%llu", first);

        // Add commas between numbers, but skip for the last term
        if (i < terms) {
            printf(", ");
        }

        // Calculate the next term and shift variables
        next = first + second;
        first = second;
        second = next;
    }

    printf("\n");
    return 0;
}
