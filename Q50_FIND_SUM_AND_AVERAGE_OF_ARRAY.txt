#include <stdio.h>

int main() {
    int n, i;
    float sum = 0.0, average;

    // Ask the user for the number of elements
    printf("Enter the number of elements in the array: ");
    scanf("%d", &n);

    // Ensure the array has at least one element to prevent division by zero
    if (n <= 0) {
        printf("Invalid array size. Please enter a number greater than 0.\n");
        return 1;
    }

    int arr[n]; // Declare an array of size n

    // Input elements from the user
    printf("Enter %d elements:\n", n);
    for (i = 0; i < n; i++) {
        printf("Element %d: ", i + 1);
        scanf("%d", &arr[i]);

        // Add each element to the running total
        sum += arr[i];
    }

    // Calculate the average (sum is a float, so fractional values are preserved)
    average = sum / n;

    // Display the final results
    printf("\n--- Results ---\n");
    printf("Sum of array elements = %.2f\n", sum);
    printf("Average of array elements = %.2f\n", average);

    return 0;
}
