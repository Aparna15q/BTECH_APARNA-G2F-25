#include <stdio.h>
#include <string.h>

void findCommonCharacters(char str1[], char str2[]) {
    // 256 covers all standard ASCII characters
    int presentInStr1[256] = {0};
    int presentInBoth[256] = {0};
    int found = 0;

    // Mark characters present in the first string
    for (int i = 0; str1[i] != '\0'; i++) {
        // Skip newline characters captured by fgets
        if (str1[i] != '\n') {
            presentInStr1[(unsigned char)str1[i]] = 1;
        }
    }

    // Check if characters from the second string exist in the first
    for (int i = 0; str2[i] != '\0'; i++) {
        if (str2[i] != '\n' && presentInStr1[(unsigned char)str2[i]] == 1) {
            presentInBoth[(unsigned char)str2[i]] = 1;
        }
    }

    // Print the common characters
    printf("Common characters: ");
    for (int i = 0; i < 256; i++) {
        if (presentInBoth[i] == 1) {
            printf("'%c' ", i);
            found = 1;
        }
    }

    if (!found) {
        printf("None");
    }
    printf("\n");
}

int main() {
    char str1[100];
    char str2[100];

    // Read string inputs safely using fgets
    printf("Enter the first string: ");
    fgets(str1, sizeof(str1), stdin);

    printf("Enter the second string: ");
    fgets(str2, sizeof(str2), stdin);

    findCommonCharacters(str1, str2);

    return 0;
}
