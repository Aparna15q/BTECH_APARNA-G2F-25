#include <stdio.h>

// Function to reverse a section of the array from 'start' to 'end' index
void reverse(int arr[], int start, int end) {
    while (start < end) {
        int temp = arr[start];
        arr[start] = arr[end];
        arr[end] = temp;
        start++;
        end--;
    }
}

// Function to right rotate the array by k positions
void rightRotate(int arr[], int n, int k) {
    // Handle cases where k is greater than the size of the array
    k = k % n;

    if (k == 0) return; // No rotation needed

    // Step 1: Reverse the entire array
    reverse(arr, 0, n - 1);

    // Step 2: Reverse the first k elements
    reverse(arr, 0, k - 1);

    // Step 3: Reverse the remaining n - k elements
    reverse(arr, k, n - 1);
}

// Function to print the array elements
void printArray(int arr[], int n) {
    for (int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
}

int main() {
    int arr[] = {1, 2, 3, 4, 5, 6, 7};
    int n = sizeof(arr) / sizeof(arr[0]);
    int k = 3; // Number of positions to rotate right

    printf("Original Array:\n");
    printArray(arr, n);

    rightRotate(arr, n, k);

    printf("Array after Right Rotation by %d positions:\n", k);
    printArray(arr, n);

    return 0;
}
