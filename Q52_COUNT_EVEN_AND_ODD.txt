#include <stdio.h>

int main() {
    int size, i;
    int even_count = 0, odd_count = 0;

    // Ask user for the size of the array
    printf("Enter the number of elements in the array: ");
    scanf("%d", &size);

    int arr[size]; // Declare array with the user-defined size

    // Input array elements from the user
    printf("Enter %d elements:\n", size);
    for (i = 0; i < size; i++) {
        printf("Element %d: ", i + 1);
        scanf("%d", &arr[i]);
    }

    // Loop through the array to check and count even and odd numbers
    for (i = 0; i < size; i++) {
        // A number is even if it is completely divisible by 2
        if (arr[i] % 2 == 0) {
            even_count++;
        } else {
            odd_count++;
        }
    }

    // Print the final counts
    printf("\nTotal Even elements: %d\n", even_count);
    printf("Total Odd elements: %d\n", odd_count);

    return 0;
}
