#include <stdio.h>

int main() {
    int arr[100];
    int size, i, j, temp;

    // 1. Get the size of the array from the user
    printf("Enter the number of elements in the array: ");
    scanf("%d", &size);

    // 2. Accept array elements
    printf("Enter %d elements:\n", size);
    for (i = 0; i < size; i++) {
        scanf("%d", &arr[i]);
    }

    // 3. Bubble Sort logic for descending order
    for (i = 0; i < size - 1; i++) {
        for (j = i + 1; j < size; j++) {
            // Swap if the current element is smaller than the next element
            if (arr[i] < arr[j]) {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }

    // 4. Print the sorted array
    printf("\nArray sorted in descending order:\n");
    for (i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");

    return 0;
}
