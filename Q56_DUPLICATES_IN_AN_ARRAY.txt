#include <stdio.h>

int main() {
    int n;

    // Get the size of the array from the user
    printf("Enter the number of elements in the array: ");
    if (scanf("%d", &n) != 1 || n <= 0) {
        printf("Invalid array size.\n");
        return 1;
    }

    int arr[n];
    int visited[n];

    // Input array elements from the user
    printf("Enter %d elements:\n", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
        visited[i] = 0; // Initialize the tracking array to 0 (unvisited)
    }

    printf("\nDuplicate elements in the array are:\n");
    int found_duplicate = 0;

    // Nested loops to find duplicates
    for (int i = 0; i < n; i++) {
        // Skip this element if it has already been processed as a duplicate
        if (visited[i] == 1) {
            continue;
        }

        int count = 1;
        for (int j = i + 1; j < n; j++) {
            if (arr[i] == arr[j]) {
                count++;
                visited[j] = 1; // Mark the identical element as visited
            }
        }

        // If the element appears more than once, print it
        if (count > 1) {
            printf("%d (appears %d times)\n", arr[i], count);
            found_duplicate = 1;
        }
    }

    if (!found_duplicate) {
        printf("No duplicate elements found.\n");
    }

    return 0;
}
