#include <stdio.h>
#include <string.h>

#define NO_OF_CHARS 256

// Function to check if two strings are anagrams
int checkAnagram(char *str1, char *str2) {
    int count[NO_OF_CHARS] = {0};
    int i;

    // If lengths are not equal, they cannot be anagrams
    if (strlen(str1) != strlen(str2)) {
        return 0;
    }

    // For each character in both strings, update the count array
    for (i = 0; str1[i] && str2[i]; i++) {
        count[(unsigned char)str1[i]]++;
        count[(unsigned char)str2[i]]--;
    }

    // If all counts are 0, then the strings are anagrams
    for (i = 0; i < NO_OF_CHARS; i++) {
        if (count[i] != 0) {
            return 0;
        }
    }

    return 1;
}

int main() {
    char str1[100], str2[100];

    printf("Enter the first string: ");
    // Read string including spaces (excluding trailing newline)
    scanf("%99[^\n]", str1);

    // Clear input buffer
    while (getchar() != '\n');

    printf("Enter the second string: ");
    scanf("%99[^\n]", str2);

    // Call function and display the result
    if (checkAnagram(str1, str2)) {
        printf("\n\"%s\" and \"%s\" are ANAGRAMS.\n", str1, str2);
    } else {
        printf("\n\"%s\" and \"%s\" are NOT ANAGRAMS.\n", str1, str2);
    }

    return 0;
}
