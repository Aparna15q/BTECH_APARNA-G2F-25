#include <stdio.h>
#include <string.h>

#define MAX_CHARS 256 // Total possible ASCII characters

// Function to find the first non-repeating character
char findFirstUniqueChar(const char *str) {
    int frequency[MAX_CHARS] = {0}; // Initialize frequency array to zero

    // Step 1: Calculate the frequency of each character
    for (int i = 0; str[i] != '\0'; i++) {
        frequency[(unsigned char)str[i]]++;
    }

    // Step 2: Traverse the string again to find the first character with a frequency of 1
    for (int i = 0; str[i] != '\0'; i++) {
        if (frequency[(unsigned char)str[i]] == 1) {
            return str[i]; // Return the first unique character found
        }
    }

    return '\0'; // Return null character if no unique character exists
}

int main() {
    char str[100];

    printf("Enter a string: ");
    // Read string including spaces (stops at newline)
    fgets(str, sizeof(str), stdin);

    // Remove trailing newline character from fgets if present
    str[strcspn(str, "\n")] = '\0';

    char result = findFirstUniqueChar(str);

    if (result != '\0') {
        printf("The first non-repeating character is: '%c'\n", result);
    } else {
        printf("All characters are repeating or the string is empty.\n");
    }

    return 0;
}
