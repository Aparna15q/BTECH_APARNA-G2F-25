#include <stdio.h>
void swap_ref(int*,int*);
int main()
int a,b;
printf("enter a & b \n");
scanf("%d%d",&a,&b);
printf("before swapping \n");
printf("a=%d \t b=%d",a,b);
swap_ref(&a,&b);
printf("\n after swapping \n");
printf("a=%d \t b=%d",a,b);
return 0;
}
void swap_ref(int*x,int*y)
{
int t=*x;
*x=*y;
*y=t;
printf("\n after swapping \n");
scanf("a=%d \t b=%d",x,y);
}
