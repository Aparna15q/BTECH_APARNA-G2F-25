#include <stdio.h>
#include <stdlib.h>

// Comparator function required by qsort to sort the array in ascending order
int compare(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}

// Function to find and print a pair with the given target sum
void findPairWithSum(int arr[], int size, int targetSum) {
    // Step 1: Sort the array first
    qsort(arr, size, sizeof(int), compare);

    // Step 2: Initialize two pointers
    int left = 0;
    int right = size - 1;
    int pairFound = 0;

    // Step 3: Traverse the array from both ends toward the middle
    while (left < right) {
        int currentSum = arr[left] + arr[right];

        if (currentSum == targetSum) {
            printf("Pair found: (%d, %d)\n", arr[left], arr[right]);
            pairFound = 1;

            // Move both pointers to find other potential pairs
            left++;
            right--;
        }
        else if (currentSum < targetSum) {
            // Sum is too small, move left pointer to increase sum
            left++;
        }
        else {
            // Sum is too large, move right pointer to decrease sum
            right--;
        }
    }

    if (!pairFound) {
        printf("No pair found with the given sum %d.\n", targetSum);
    }
}

int main() {
    int arr[] = {8, 7, 2, 5, 3, 1};
    int targetSum = 10;
    int size = sizeof(arr) / sizeof(arr[0]);

    printf("Array elements: ");
    for (int i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }
    printf("\nTarget Sum: %d\n\n", targetSum);

    findPairWithSum(arr, size, targetSum);

    return 0;
}
