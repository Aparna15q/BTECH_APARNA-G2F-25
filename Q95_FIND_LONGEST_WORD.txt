#include <stdio.h>
#include <string.h>
#include <ctype.h>

#define MAX_LIMIT 200

int main() {
    char sentence[MAX_LIMIT];

    printf("Enter a sentence: ");
    // Use fgets to safely read a line including spaces
    if (fgets(sentence, sizeof(sentence), stdin) == NULL) {
        return 1;
    }

    int i = 0;
    int current_len = 0;
    int max_len = 0;
    int start_index = 0;
    int max_start_index = 0;

    while (sentence[i] != '\0') {
        // Check if the current character is an alphanumeric word character
        if (isalnum((unsigned char)sentence[i])) {
            if (current_len == 0) {
                start_index = i; // Mark the beginning of a new word
            }
            current_len++;
        } else {
            // End of a word reached; compare its length with the maximum found so far
            if (current_len > max_len) {
                max_len = current_len;
                max_start_index = start_index;
            }
            current_len = 0; // Reset counter for the next word
        }
        i++;
    }

    // Final check in case the sentence ends directly on a word without trailing punctuation/space
    if (current_len > max_len) {
        max_len = current_len;
        max_start_index = start_index;
    }

    // Print the results safely
    printf("\nResults:\n");
    if (max_len > 0) {
        printf("Longest word: ");
        // Print exactly 'max_len' characters starting from 'max_start_index'
        for (int j = 0; j < max_len; j++) {
            putchar(sentence[max_start_index + j]);
        }
        printf("\nLength: %d\n", max_len);
    } else {
        printf("No valid words found.\n");
    }

    return 0;
}
