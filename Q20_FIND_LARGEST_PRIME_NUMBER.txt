#include <stdio.h>
#include <math.h>

// Function to check if a number is prime
int isPrime(int n) {
    // Numbers less than or equal to 1 are not prime
    if (n <= 1) return 0;
    // 2 is the only even prime number
    if (n == 2) return 1;
    // Exclude all other even numbers
    if (n % 2 == 0) return 0;

    // Check odd factors up to the square root of n
    int limit = (int)sqrt(n);
    for (int i = 3; i <= limit; i += 2) {
        if (n % i == 0) {
            return 0; // Found a factor, so not prime
        }
    }
    return 1; // Number is prime
}

int main() {
    int limit;

    printf("Enter the upper limit to find the largest prime number: ");
    if (scanf("%d", &limit) != 1) {
        printf("Invalid input.\n");
        return 1;
    }

    // Search downwards from the limit to find the largest prime
    int largestPrime = -1;
    for (int i = limit; i >= 2; i--) {
        if (isPrime(i)) {
            largestPrime = i;
            break; // Stop immediately upon finding the highest prime
        }
    }

    // Output the result
    if (largestPrime != -1) {
        printf("The largest prime number up to %d is: %d\n", limit, largestPrime);
    } else {
        printf("There are no prime numbers up to %d.\n", limit);
    }

    return 0;
}
