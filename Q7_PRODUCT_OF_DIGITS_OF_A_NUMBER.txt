#include <stdio.h>
#include <stdlib.h> // For abs() function

int main() {
    int num, temp, rem;
    long long product = 1; // Used long long to handle larger products

    // Take input from the user
    printf("Enter any number: ");
    scanf("%d", &num);

    // Store the original number for the final print statement
    temp = num;

    // Handle the special case where the input number is 0
    if (num == 0) {
        product = 0;
    } else {
        // Convert negative numbers to positive to process digits properly
        num = abs(num);

        // Loop to extract and multiply digits
        while (num > 0) {
            rem = num % 10;      // Extract the last digit
            product = product * rem; // Multiply it with the running product
            num = num / 10;      // Remove the last digit from the number
        }
    }

    // Display the result
    printf("The product of the digits of %d is: %lld\n", temp, product);

    return 0;
}
