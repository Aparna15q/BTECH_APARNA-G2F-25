#include <stdio.h>

int main() {
    int rows, i, j, space;

    printf("Enter the number of rows: ");
    scanf("%d", &rows);

    for (i = 1; i <= rows; i++) {
        // Loop to print leading spaces for centering
        for (space = 1; space <= rows - i; space++) {
            printf("  ");
        }

        // Loop to print the increasing numbers
        for (j = 1; j <= i; j++) {
            printf("%d ", j);
        }

        // Loop to print the decreasing numbers
        for (j = i - 1; j >= 1; j--) {
            printf("%d ", j);
        }

        // Move to the next line after completing a row
        printf("\n"
);
    }

    return 0;
}
