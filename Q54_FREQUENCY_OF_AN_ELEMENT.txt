#include <stdio.h>

int main() {
    int n, i, j, count;

    // Get the size of the array from the user
    printf("Enter the number of elements: ");
    scanf("%d", &n);

    int arr[n];
    int visited[n];

    // Read array elements and initialize the visited array
    printf("Enter %d elements:\n", n);
    for(i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
        visited[i] = 0; // 0 indicates the element has not been counted yet
    }

    // Find and print the frequency of each element
    printf("\nFrequency of each element:\n");
    for(i = 0; i < n; i++) {
        // Skip if the element has already been counted
        if(visited[i] == 1) {
            continue;
        }

        count = 1;
        for(j = i + 1; j < n; j++) {
            if(arr[i] == arr[j]) {
                count++;
                visited[j] = 1; // Mark the duplicate element as counted
            }
        }

        printf("%d occurs %d times\n", arr[i], count);
    }

    return 0;
}
