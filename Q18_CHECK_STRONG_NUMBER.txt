#include <stdio.h>

// Function to calculate factorial of a digit
int getFactorial(int n) {
    int fact = 1;
    for (int i = 1; i <= n; i++) {
        fact *= i;
    }
    return fact;
}

int main() {
    int num, temp, digit;
    int factorial_sum = 0;

    printf("Enter a number: ");
    if (scanf("%d", &num) != 1) return 1;

    temp = num;

    // Process each digit
    while (temp > 0) {
        digit = temp % 10;
        factorial_sum += getFactorial(digit);
        temp /= 10;
    }

    // Verify the Strong Number condition
    if (factorial_sum == num) {
        printf("%d is a Strong Number.\n", num);
    } else {
        printf("%d is not a Strong Number.\n", num);
    }

    return 0;
}
