#include <stdio.h>

int main() {
    int rows, i, j;

    // Prompt user for input
    printf("Enter the number of rows: ");
    scanf("%d", &rows);

    // Outer loop for rows (counting down)
    for (i = rows; i >= 1; i--) {
        // Inner loop to print numbers in each row
        for (j = 1; j <= i; j++) {
            printf("%d ", j);
        }
        // Move to the next line
        printf("\n");
    }

    return 0;
}
