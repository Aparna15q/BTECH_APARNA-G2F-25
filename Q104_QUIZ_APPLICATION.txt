#include <stdio.h>
#include <stdlib.h>
#include <ctype.h>

#define TOTAL_QUESTIONS 5

// Structure to store a single quiz question
typedef struct {
    char question_text[256];
    char options[4][64];
    char correct_option; // 'A', 'B', 'C', or 'D'
} QuizQuestion;

// Function declarations
void display_welcome_menu();
void run_quiz(QuizQuestion quiz[]);
void display_scorecard(int final_score, int total_questions);

int main() {
    // Initialize question bank
    QuizQuestion bank[TOTAL_QUESTIONS] = {
        {
            "Which programming language is known as the mother of all languages?",
            {"A. Python", "B. Java", "C. C Language", "D. C++"},
            'C'
        },
        {
            "What is the only format specifier used to print a single character in C?",
            {"A. %s", "B. %c", "C. %d", "D. %f"},
            'B'
        },
        {
            "Which of the following is not a valid loop structure in C?",
            {"A. for", "B. while", "C. do-while", "D. foreach"},
            'D'
        },
        {
            "What is the size of an 'int' data type in standard 32-bit C compilers?",
            {"A. 1 Byte", "B. 2 Bytes", "C. 4 Bytes", "D. 8 Bytes"},
            'C'
        },
        {
            "Which keyword is used to return a value from a function?",
            {"A. break", "B. return", "C. continue", "D. exit"},
            'B'
        }
    };

    char choice;

    do {
        display_welcome_menu();
        printf("Enter your choice: ");
        scanf(" %c", &choice);
        choice = toupper(choice);

        switch (choice) {
            case 'S':
                run_quiz(bank);
                break;
            case 'Q':
                printf("\nThank you for using the Quiz Application. Goodbye!\n");
                break;
            default:
                printf("\nInvalid option! Please try again.\n");
        }
    } while (choice != 'Q');

    return 0;
}

// Displays the initial home screen terminal interface
void display_welcome_menu() {
    printf("\n==================================\n");
    printf("    WELCOME TO THE C QUIZ APP     \n");
    printf("==================================\n");
    printf("[S] Start Quiz\n");
    printf("[Q] Quit Application\n");
    printf("----------------------------------\n");
}

// Manages the gameplay loop, fetches input, and calculates scoring
void run_quiz(QuizQuestion quiz[]) {
    int score = 0;
    char user_answer;

    printf("\n--- Quiz Started! Best of Luck! ---\n");

    for (int i = 0; i < TOTAL_QUESTIONS; i++) {
        printf("\nQuestion %d: %s\n", i + 1, quiz[i].question_text);

        // Print the 4 multiple choice options
        for (int j = 0; j < 4; j++) {
            printf("%s\n", quiz[i].options[j]);
        }

        // Handle user input safely
        printf("Your Answer (A/B/C/D): ");
        scanf(" %c", &user_answer);
        user_answer = toupper(user_answer); // Normalize input to uppercase

        // Check if correct
        if (user_answer == quiz[i].correct_option) {
            printf("✅ Correct!\n");
            score++;
        } else {
            printf("❌ Wrong! The correct answer was %c.\n", quiz[i].correct_option);
        }
    }

    display_scorecard(score, TOTAL_QUESTIONS);
}

// Renders the final performance report block
void display_scorecard(int final_score, int total_questions) {
    float percentage = ((float)final_score / total_questions) * 100;

    printf("\n==================================\n");
    printf("           QUIZ COMPLETED         \n");
    printf("==================================\n");
    printf("Final Score: %d out of %d\n", final_score, total_questions);
    printf("Percentage:  %.2f%%\n", percentage);
    printf("==================================\n");
}
